CREATE TABLE hotel(
uniqueID int NOT NULL,
name VARCHAR(30),
lat DOUBLE(8, 8),
lng DOUBLE(8, 8),
address TEXT,
description TEXT,
price INT,
PRIMARY KEY(uniqueID));
